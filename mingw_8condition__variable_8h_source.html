65  bool wait_impl(M& lock, DWORD timeout)
66  {
67  {
68  lock_guard<recursive_mutex> guard(mMutex);
69  mNumWaiters++;
70  }
71  lock.unlock();
72  DWORD ret = WaitForSingleObject(mSemaphore, timeout);
73 
74  mNumWaiters--;
75  SetEvent(mWakeEvent);
76  lock.lock();
77  if (ret == WAIT_OBJECT_0)
78  return true;
79  else if (ret == WAIT_TIMEOUT)
80  return false;

82 //1)The point in notify_all() where we determine the count to
83 //increment the semaphore with has not been reached yet:
84 //we just need to decrement mNumWaiters, but setting the event does not hurt
85 //
86 //2)Semaphore has just been released with mNumWaiters just before
87 //we decremented it. This means that the semaphore count

110  void notify_all() noexcept
111  {
112  lock_guard<recursive_mutex> lock(mMutex); //block any further wait requests until all current waiters are unblocked
113  if (mNumWaiters.load() <= 0)
114  return;
115 
116  ReleaseSemaphore(mSemaphore, mNumWaiters, NULL);
117  while(mNumWaiters > 0)
118  {
119  auto ret = WaitForSingleObject(mWakeEvent, 1000);
120  if ((ret == WAIT_FAILED) || (ret == WAIT_ABANDONED))
121  throw std::system_error(EPROTO, std::generic_category());
122  }
123  assert(mNumWaiters == 0);
124 //in case some of the waiters timed out just after we released the
125 //semaphore by mNumWaiters, it won't be zero now, because not all waiters
126 //woke up by acquiring the semaphore. So we must zero the semaphore before
127 //we accept waiters for the next event
128 //See _wait_impl for details
129  while(WaitForSingleObject(mSemaphore, 0) == WAIT_OBJECT_0);
130  }
131  void notify_one() noexcept
132  {
133  lock_guard<recursive_mutex> lock(mMutex);
134  if (!mNumWaiters)
135  return;
136  int targetWaiters = mNumWaiters.load() - 1;
137  ReleaseSemaphore(mSemaphore, 1, NULL);
138  while(mNumWaiters > targetWaiters)

143  }
