120  char min[MSG_SIZE];
121  char cls[MSG_SIZE];
122  char buffer[MSG_SIZE];
123 
124  /* Get descriptions for the major and minor error numbers */
125  if (H5Eget_class_name(err_desc->cls_id, cls, MSG_SIZE) < 0)
126  return -1;
127

132  return -1;
133

135 #if (defined __APPLE__) // not even tested to compile this
136  snprintf(buffer, MSG_SIZE, fmt, cls, maj, min);

139 #elif (defined __GNUG__)
140  snprintf(buffer, MSG_SIZE, fmt, cls, maj, min);
141 #endif
142  buffer[MSG_SIZE - 1] = 0; // its the only way to be sure ...
143  try
144  {
145  out->append(buffer);
146  }
147  catch (...)
