106 static herr_t custom_print_cb(unsigned n, const H5E_error2_t *err_desc, void* client_data)
107 {
108  const int MSG_SIZE = 256;
109  auto *out = static_cast<std::string*>(client_data);
110  char maj[MSG_SIZE];
111  char min[MSG_SIZE];
112  char cls[MSG_SIZE];
113  char buffer[MSG_SIZE];
114 
115  /* Get descriptions for the major and minor error numbers */
116  if (H5Eget_class_name(err_desc->cls_id, cls, MSG_SIZE) < 0)
117  return -1;
118 
119  if (H5Eget_msg(err_desc->maj_num, NULL, maj, MSG_SIZE)<0)
120  return -1;
121 
122  if (H5Eget_msg(err_desc->min_num, NULL, min, MSG_SIZE)<0)
123  return -1;
124 
125  const char* fmt = " * %s %s: %s";
126 #if (defined __APPLE__) // not even tested to compile this
127  snprintf(buffer, MSG_SIZE, fmt, cls, maj, min);
128 #elif (defined _MSC_VER)
129  sprintf_s(buffer, MSG_SIZE, fmt, cls, maj, min);
130 #elif (defined __GNUG__)
131  snprintf(buffer, MSG_SIZE, fmt, cls, maj, min);
132 #endif
133  buffer[MSG_SIZE - 1] = 0; // its the only way to be sure ...
134  try
135  {
136  out->append(buffer);
137  }
138  catch (...)
139  {
140  // suppress all exceptions since thsi function is called from a c library.
141  }
142  return 0;
143 }
